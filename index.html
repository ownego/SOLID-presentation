S.O.L.I.D

Single Responsibility Principle

A module should have one, and only one, reason to change

Single Responsiblity Principle

Before
							
                                class ChessBoard {
                                    protected $board;
                                    public function __construct($size){}
                                    public function placeQueen($x, $y){}
                                    public function renderToScreen(){}
                                    public function renderToJSON(){}
                                }

Single Responsiblity Principle

After
							
                                class ChessBoard {
                                    protected $board;
                                    public function __construct($size){}
                                    public function placeQueen($x, $y){}
                                }
                            
						
							
                                class ChessBoardRenderer {
                                    protected $chessBoard;
                                    public function __construct(ChessBoard $chessBoard){}
                                    public function renderToScreen(){}
                                    public function renderToJSON(){}
                                }

Open-Closed Principle

A software artifact should be open for extension but closed for modification

Open-closed principle

Before
							
                                class ChessBoard {
                                    protected $board;
                                    public function __construct($size){}
                                    public function placeQueen($x, $y){}
                                }
                            
                        
							
                                class ChessBoardRenderer {
                                    protected $chessBoard;
                                    public function __construct(ChessBoard $chessBoard){}
                                    public function renderToScreen(){}
                                    public function renderToJSON(){}
                                }

Open-closed principle

After
							
                                class ChessBoard {
                                    protected $board;
                                    public function __construct($size){}
                                    public function placeQueen($x, $y){}
                                }
                            
                        
							
                                interface ChessBoardRenderer
                                {
                                    public function render();
                                }

Open-closed principle

After
							
                                class StandardChessBoardRenderer implements ChessBoardRenderer {
                                    protected $chessBoard;
                                    public function __construct(ChessBoard $chessBoard){}
                                    public function render(){}
                                }
                            
                        
							
                                class JSONChessBoardRenderer implements ChessBoardRenderer {
                                    protected $chessBoard;
                                    public function __construct(ChessBoard $chessBoard){}
                                    public function render(){}
                                }

Liskov Substitution Principle

What is wanted here is something like the following substitution property: If for each object o1 of type S there is an object o2 of type T such that for all programs P defined in terms of T, the behavior of P is unchanged when o1 is substituted for o2 then S is a subtype of T

Liskov Substitution Principle

Before 
                            
                                class Rectangle {
                                    protected $width;
                                    protected $height;
                                    public function __construct($width, $height){}
                                    public function area(){}
                                }
                                
                                class Square extends Rectangle {}

                                class Main {
                                    public static function getArea(Rectangle $rectangle) {
                                        return $rectangle->area();
                                    }
                                }

Liskov Substitution Principle

After 
                            
                                interface CalculateArea
                                {
                                    public function area();
                                }

Liskov Substitution Principle

After 
                            
                                class Rectangle implements CalculateArea {
                                    protected $width;
                                    protected $height;
                                    public function __construct($width, $height){}
                                    public function area(){}

                                }
                            
                        
                            
                                class Square implements CalculateArea {
                                    protected $side;
                                    public function __construct($side){}
                                    public function area(){}
                                }

Liskov Substitution Principle

After 
                            
                                class Main {
                                    public static function getArea(CalculateArea $shape) {
                                        return $shape->area();
                                    }
                                }

Dependency Inversion Principle

Dependency Inversion Principle

Before 
                            
                                class Controller {
                                    public function __construct(){}
                                    public function save()
                                    {
                                        $db = new MysqlDB();
                                        $db->save($this);
                                    }
                                }

Dependency Inversion Principle

After 
                            
                                class Controller {
                                    public function __construct(DB $db){}
                                    public function save() {
                                        $this->db->save($this);
                                    }
                                }
